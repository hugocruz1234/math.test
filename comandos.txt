-- Seleccionar achivos para subir
 
git add "Nombre del archivo" o git add . (punto para agregar todos los archivos)


--Agregar el commit
git commit -m ""


--Subir archivos a git
git push


-- Renombrar o mover de carpetas archivos
git mv [nombre archivo a editar] [nombre nuevo]

-- Crear una nueva rama y cambiarse a ella
git checkout -b [nombre rama]

--Reiniciar VSCode en el directorio en el que estamos
code -r .

-- Jalar cambios de main
git pull [nombre remote] [nombre rama]
